20       */
21      public DefaultCachedHttpResponse(final ChannelBuffer channelBuffer,
22          final ChannelFutureListener channelFutureListener) {
23          this.channelBuffer = channelBuffer;
24          this.channelFutureListener = channelFutureListener;
25      }
26  
27      public ChannelBuffer getChannelBuffer() {
28          // We can never return the original buffer because multiple threads 
29          // could then access it and modify the mutable data.
30          // NOTE: This does not copy the actual bytes.
31          return channelBuffer.duplicate();
32      }
