127     /***
128      * Strips the host from a URI string. This will turn "http://host.com/path"
129      * into "/path".
130      * 
131      * @param uri The URI to transform.
132      * @return A string with the URI stripped.
133      */
134     public static String stripHost(final String uri) {
135         if (!uri.startsWith("http")) {
136             // It's likely a URI path, not the full URI (i.e. the host is 
137             // already stripped).
138             return uri;
139         }
140         final String noHttpUri = StringUtils.substringAfter(uri, "://");
141         final int slashIndex = noHttpUri.indexOf("/");
142         if (slashIndex == -1) {
143             return "/";
144         }
145         final String noHostUri = noHttpUri.substring(slashIndex);
146         return noHostUri;
147     }

149     /***
150      * Builds the cache URI from the request, including the host and the path.
151      * 
152      * @param httpRequest The request.
153      * @return The cache URI.
154      */
155     public static String cacheUri(final HttpRequest httpRequest) {
156         final String host = httpRequest.getHeader(HttpHeaders.Names.HOST);
157         final String uri = httpRequest.getUri();
158         final String path;
159         if (uri.startsWith("http")) {
160             path = stripHost(uri);
161         }
162         else {
163             path = uri;
164         }
165         
166         // TODO: We don't append http:// or https:// here. Is it really OK to
167         // ignore the protocol?
168         return host + path;
169     }

218      * Copies the mutable fields from the response original to the copy.
219      * 
220      * @param original The original response to copy from.
221      * @param copy The copy.
222      * @return The copy with all mutable fields from the original.
223      */
224     public static HttpResponse copyMutableResponseFields(
225         final HttpResponse original, final HttpResponse copy) {
226         
227         final Collection<String> headerNames = original.getHeaderNames();
228         for (final String name : headerNames) {
229             final List<String> values = original.getHeaders(name);
230             copy.setHeader(name, values);
231         }
232         copy.setContent(original.getContent());
233         if (original.isChunked()) {
234             copy.setChunked(true);
235         }
236         return copy;
237     }

252     /***
253      * Writes a raw HTTP response to the channel. 
254      * 
255      * @param channel The channel.
256      * @param statusLine The status line of the response.
257      * @param headers The raw headers string.
258      * @param responseBody The response body.
259      */
260     public static void writeResponse(final Channel channel, 
261         final String statusLine, final String headers, 
262         final String responseBody) {
263         final String fullResponse = statusLine + headers + responseBody;
264         LOG.info("Writing full response:\n"+fullResponse);
265         try {
266             final ChannelBuffer buf = 
267                 ChannelBuffers.copiedBuffer(fullResponse.getBytes("UTF-8"));
268             channel.write(buf);
269             channel.setReadable(true);
270             return;
271         }
272         catch (final UnsupportedEncodingException e) {
273             // Never.
274             return;
275         }    
276     }

340     /***
341      * Parses the host and port an HTTP request is being sent to.
342      * 
343      * @param httpRequest The request.
344      * @return The host and port string.
345      */
346     public static String parseHostAndPort(final HttpRequest httpRequest) {
347         final String uri = httpRequest.getUri();
348         final String tempUri;
349         if (!uri.startsWith("http")) {
350             // Browsers particularly seem to send requests in this form when
351             // they use CONNECT.
352             tempUri = uri;
353         }
354         else {
355             // We can't just take a substring from a hard-coded index because it
356             // could be either http or https.
357             tempUri = StringUtils.substringAfter(uri, "://");
358         }
359         final String hostAndPort;
360         if (tempUri.contains("/")) {
361             hostAndPort = tempUri.substring(0, tempUri.indexOf("/"));
362         }
363         else {
364             hostAndPort = tempUri;
365         }
366         return hostAndPort;
367     }

369     /***
370      * Parses the port from an address.
371      * 
372      * @param httpRequest The request containing the URI.
373      * @return The port. If not port is explicitly specified, returns the 
374      * the default port 80 if the protocol is HTTP and 443 if the protocol is
375      * HTTPS.
376      */
377     public static int parsePort(final HttpRequest httpRequest) {
378         final String uri = httpRequest.getUri();
379         if (uri.contains(":")) {
380             final String portStr = StringUtils.substringAfter(uri, ":"); 
381             return Integer.parseInt(portStr);
382         }
383         else if (uri.startsWith("http")) {
384             return 80;
385         }
386         else if (uri.startsWith("https")) {
387             return 443;
388         }
389         else {
390             // Unsupported protocol -- return 80 for now.
391             return 80;
392         }
393     }

395     /***
396      * Creates a copy of an original HTTP request to void modifying it.
397      * 
398      * @param original The original request.
399      * @param keepProxyFormat keep proxy-formatted URI (used in chaining) 
400      * @return The request copy.
401      */
402     public static HttpRequest copyHttpRequest(final HttpRequest original, 
403         boolean keepProxyFormat) {
404         final HttpMethod method = original.getMethod();
405         final String uri = original.getUri();
406         LOG.info("Raw URI before switching from proxy format: {}", uri);
407         final HttpRequest copy;
408 
409         if (keepProxyFormat) {
410             copy = new DefaultHttpRequest(original.getProtocolVersion(), 
411                 method, uri);
412         } else {
413             final String noHostUri = ProxyUtils.stripHost(uri);
414             copy = new DefaultHttpRequest(original.getProtocolVersion(), 
415                 method, noHostUri);
416         }
417         
418         final ChannelBuffer originalContent = original.getContent();
419         
420         if (originalContent != null) {
421             copy.setContent(originalContent);
422         }
423         
424         // We also need to follow 2616 section 13.5.1 End-to-end and 
425         // Hop-by-hop Headers
426         // The following HTTP/1.1 headers are hop-by-hop headers:
427         //   - Connection
428         //   - Keep-Alive
429         //   - Proxy-Authenticate
430         //   - Proxy-Authorization
431         //   - TE
432         //   - Trailers
433         //   - Transfer-Encoding
434         //   - Upgrade
435         
436         LOG.info("Request copy method: {}", copy.getMethod());
437         copyHeaders(original, copy);
438 
439         final String ae = copy.getHeader(HttpHeaders.Names.ACCEPT_ENCODING);
440         if (StringUtils.isNotBlank(ae)) {
441             // Remove sdch from encodings we accept since we can't decode it.
442             final String noSdch = ae.replace(",sdch", "").replace("sdch", "");
443             copy.setHeader(HttpHeaders.Names.ACCEPT_ENCODING, noSdch);
444             LOG.info("Removed sdch and inserted: {}", noSdch);
445         }
446         
447         // Switch the de-facto standard "Proxy-Connection" header to 
448         // "Connection" when we pass it along to the remote host. This is 
449         // largely undocumented but seems to be what most browsers and servers
450         // expect.
451         final String proxyConnectionKey = "Proxy-Connection";
452         if (copy.containsHeader(proxyConnectionKey)) {
453             final String header = copy.getHeader(proxyConnectionKey);
454             copy.removeHeader(proxyConnectionKey);
455             copy.setHeader("Connection", header);
456         }
457         
458         ProxyUtils.addVia(copy);
459         return copy;
460     }

462     private static void copyHeaders(final HttpMessage original, 
463         final HttpMessage copy) {
464         final Set<String> headerNames = original.getHeaderNames();
465         for (final String name : headerNames) {
466             if (!hopByHopHeaders.contains(name.toLowerCase())) {
467                 final List<String> values = original.getHeaders(name);
468                 copy.setHeader(name, values);
469             }
470         }
471     }

474      * Removes all headers that should not be forwarded.
475      * See RFC 2616 13.5.1 End-to-end and Hop-by-hop Headers.

477      * @param msg The message to strip headers from.

479     public static void stripHopByHopHeaders(final HttpMessage msg) {

481         for (final String name : headerNames) {
482             if (hopByHopHeaders.contains(name.toLowerCase())) {
483                 msg.removeHeader(name);

485         }

499     /***
500      * Adds the Via header to specify that the message has passed through
501      * the proxy.
502      * 
503      * @param msg The HTTP message.
504      */
505     public static void addVia(final HttpMessage msg) {
506         final StringBuilder sb = new StringBuilder();
507         sb.append(msg.getProtocolVersion().getMajorVersion());
508         sb.append(".");
509         sb.append(msg.getProtocolVersion().getMinorVersion());
510         sb.append(".");
511         sb.append(hostName);
512         final List<String> vias; 
513         if (msg.containsHeader(HttpHeaders.Names.VIA)) {
514             vias = msg.getHeaders(HttpHeaders.Names.VIA);
515             vias.add(sb.toString());
516         }
517         else {
518             vias = Arrays.asList(sb.toString());
519         }
520         msg.setHeader(HttpHeaders.Names.VIA, vias);
521     }
